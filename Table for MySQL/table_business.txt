Series_reference VARCHAR(255) NOT NULL, 
Period INT NOT NULL,

UNITS VARCHAR(255),
Subject VARCHAR(255), 
‘Group’ VARCHAR(255), 
Series_title_1 VARCHAR(255),

Series_title_3 VARCHAR(255), 
Series_title_4 VARCHAR(255), 
Series_title_5 VARCHAR(255), 
id INT NOT NULL AUTO_INCREMENT,
